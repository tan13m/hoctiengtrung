["liǎng"] : helper(wan); res = wanPinyin; res.push(pinyinData["wan"]); } if (remainder > 0) { if (wan > 0 && remainder < 1000) res.push(pinyinData[0]); res = res.concat(helper(remainder)); } return res; }
